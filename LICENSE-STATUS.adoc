grep -r "SPDX-License-Identifier: AGPL-3.0-or-later" --include="*.rs" --include="*.zig" --include="*.idr" --include="*.sh" --include="*.toml" --include="*.yml" --include="*.yaml" --include="*.json" .

grep -r "SPDX-License-Identifier: PMPL" --include="*.rs" --include="*.zig" --include="*.idr" --include="*.sh" --include="*.toml" . | wc -l
